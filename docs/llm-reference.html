1. What Is Clinvio Nucleus

Clinvio Nucleus is a 23-module enterprise Java framework for building web applications with:

  • Server-side rendering via Thymeleaf + HTMX (no JavaScript frameworks)
  • AES-256-GCM field-level encryption for sensitive data
  • SQLite database (default) with optional PostgreSQL migration
  • JWT authentication, GDPR compliance, workflow engine
  • Observability, caching, scheduling, email, notifications

4. Core Concepts

4.1 Entity Inheritance

All entities must extend BaseEntity which provides:

  • id (Long) — auto-generated primary key
  • createdAt (LocalDateTime) — set on persist
  • updatedAt (LocalDateTime) — set on update
  • deletedAt (LocalDateTime) — soft delete (null = active)
import hu.clinvio.ui.persistence.entity.BaseEntity;

@Entity
@Table(name = "orders")
public class Order extends BaseEntity {
    private String orderNumber;
    // id, createdAt, updatedAt, deletedAt are inherited
}

4.2 Field Encryption

Mark String fields with @SensitiveData + @Convert for AES-256-GCM encryption.

IMPORTANT: JPA does not process @Convert on meta-annotations. You MUST add both annotations.
import hu.clinvio.ui.persistence.annotation.SensitiveData;
import hu.clinvio.ui.persistence.crypto.EncryptedStringConverter;

@SensitiveData
@Convert(converter = EncryptedStringConverter.class)
@Column(nullable = false)
private String customerName;  // Encrypted in database

private String orderNumber;   // NOT encrypted
WARNING: Encrypted fields CANNOT be searched with SQL LIKE queries. Search only on non-encrypted fields.

4.3 Repository Pattern

import hu.clinvio.ui.business.repository.CvBaseRepository;

@Repository
public interface OrderRepository extends CvBaseRepository<Order, Long> {
    // CvBaseRepository provides: findAllActive(), findActiveById(), softDelete(), countActive()
    
    @Query("SELECT o FROM Order o WHERE o.deletedAt IS NULL AND o.orderNumber LIKE %:q%")
    Page<Order> search(@Param("q") String query, Pageable pageable);
}

4.4 Service Pattern

import hu.clinvio.ui.business.service.CvSearchableService;

@Service
@Transactional(readOnly = true)
public class OrderService extends CvSearchableService<Order, Long> {

    public OrderService(OrderRepository repo) {
        super(repo, repo);  // First arg: CvBaseRepository, Second: JpaSpecificationExecutor
    }

    @Override
    public Page<Order> search(String query, Pageable pageable) {
        if (query == null || query.isBlank()) return findAll(pageable);
        return ((OrderRepository) repository).search(query, pageable);
    }

    @Override
    protected String getEntityName() { return "Order"; }
}

// CvBaseService provides: findAll(), findById(), getById(), create(), update(), delete(), count()
// CvSearchableService adds: search(), findBySpec(), countBySpec()

4.5 Controller Pattern

import hu.clinvio.ui.business.controller.CvBaseCrudController;

@Controller
@RequestMapping("/orders")
public class OrderController extends CvBaseCrudController<Order> {

    public OrderController(OrderService service) {
        super(service);
    }

    @Override protected String getListTemplate() { return "orders/list"; }
    @Override protected String getDetailTemplate() { return "orders/detail"; }
    @Override protected String getEntityName() { return "Order"; }
    @Override protected String getEntityPath() { return "/orders"; }

    @Override
    protected CvDataTable<Order> buildDataTable(Page<Order> page) {
        return new CvDataTable<Order>()
            .addColumn("orderNumber", "Order #", true)
            .addColumn("customerName", "Customer", true)
            .data(page.getContent())
            .searchable(true).pageable(true).pageSize(page.getSize());
    }

    // YOU must add @GetMapping to listView and detailView:
    @GetMapping
    public String listView(Model model, Pageable pageable) {
        return super.listView(model, pageable);  // or custom dashboard
    }

    @GetMapping("/{id}")
    public String detailView(@PathVariable Long id, Model model) {
        return super.detailView(id, model);
    }
}
NOTE: CvBaseCrudController does NOT have @GetMapping on view methods. Subclasses must add their own mappings. This prevents conflicts when overriding with a custom dashboard.

5. Business Features

5.1 Workflow Engine

import hu.clinvio.ui.business.workflow.WorkflowEngine;

WorkflowEngine<OrderStatus> workflow = WorkflowEngine.of(OrderStatus.class)
    .transition(PENDING, PROCESSING, CANCELLED)
    .transition(PROCESSING, SHIPPED, CANCELLED)
    .transition(SHIPPED, DELIVERED)
    .onTransition(PROCESSING, SHIPPED, (from, to) -> {
        // callback on this transition
    })
    .build();

workflow.validate(currentStatus, newStatus);  // throws CvBusinessException if invalid
workflow.transition(order, newStatus, order::setStatus);  // with callbacks

import hu.clinvio.ui.business.util.CvValidation;
import hu.clinvio.ui.business.validation.CvValidationResult;

CvValidationResult result = CvValidation.builder()
    .require(name, "name", "Name is required")
    .requireEmail(email, "email")
    .requireMinLength(password, 8, "password")
    .build();

if (result.hasErrors()) {
    throw new CvValidationException("Invalid input", result);
